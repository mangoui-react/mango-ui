## Installation

### Global Installation

<PackageManagers
  command={{
    npm: 'npm i @melio-ui/react',
    yarn: 'yarn add @melio-ui/react',
    pnpm: 'pnpm add @melio-ui/react',
    bun: 'bun add @melio-ui/react',
  }}
/>

### Individual Installation

<PackageManagers
  command={{
    npm: 'npm i @melio-ui/accordion',
    yarn: 'yarn add @melio-ui/accordion',
    pnpm: 'pnpm add @melio-ui/accordion',
    bun: 'bun add @melio-ui/accordion',
  }}
/>

## Import

<ImportSyntax
  command={{
    individual: `import { Accordion } from '@melio-ui/accordion';

export default function App() {
  return (
    <Accordion.Root>
    ...
    </Accordion.Root>
  );
}
`,
    global: `import { Accordion } from '@melio-ui/react';

export default function App() {
  return (
    <Accordion.Root>
    ...
    </Accordion.Root>
  );
}
`,
}}
/>
